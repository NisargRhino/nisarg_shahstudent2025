    <script>
        function visualizeMolecule() {
            const smiles = document.getElementById('smilesInput').value;
            if (!smiles) {
                alert('Please enter a SMILES string.');
                return;
            }

            const element = document.getElementById('viewer');
            const viewer = $3Dmol.createViewer(element, {
                backgroundColor: '#fff'
            });

            const options = {
                width: 600,
                height: 400,
                resizable: true,
                input_format: 'smi'
            };

            const url = `https://3dmol.csb.pitt.edu/3Dmol/convert?smiles=${encodeURIComponent(smiles)}&format=mol`;

            $3Dmol.download(url, viewer, {}, function() {
                viewer.setStyle({}, {stick: {}});
                viewer.zoomTo();
                viewer.render();
            });
        }
    </script>
