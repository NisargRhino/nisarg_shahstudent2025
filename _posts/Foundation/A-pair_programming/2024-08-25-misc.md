    <script>
        function visualizeMolecule() {
            const element = document.getElementById('viewer');
            const viewer = $3Dmol.createViewer(element, {
                backgroundColor: '#fff'
            });

            // Clear the viewer before rendering a new molecule
            viewer.clear();

            // Load an example PDB structure (methane)
            const pdbData = `HETATM    1  C   UNK     1       0.000   0.000   0.000  1.00  0.00           C  
                            HETATM    2  H1  UNK     1       0.629   0.629   0.629  1.00  0.00           H  
                            HETATM    3  H2  UNK     1      -0.629  -0.629   0.629  1.00  0.00           H  
                            HETATM    4  H3  UNK     1      -0.629   0.629  -0.629  1.00  0.00           H  
                            HETATM    5  H4  UNK     1       0.629  -0.629  -0.629  1.00  0.00           H  
                            END`;

            // Add PDB data to viewer
            viewer.addModel(pdbData, "pdb");

            // Set the style and render
            viewer.setStyle({}, {stick: {}});
            viewer.zoomTo();
            viewer.render();
        }
    </script>
